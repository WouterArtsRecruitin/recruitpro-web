## 🔧 PIPEDRIVE AUTOMATION SETUP

### **LEAD SCORING SYSTEEM**

```javascript
// Pipedrive Custom Fields Setup
const customFields = {
    // Assessment Data
    assessment_score: 'number', // 0-100
    maturity_level: 'varchar', // Starter/Basis/Gemiddeld/Gevorderd/Expert
    assessment_date: 'date',
    
    // Lead Scoring
    lead_score: 'number', // 0-100 calculated score
    priority_level: 'varchar', // Hot/Warm/Cold
    
    // Qualification Data
    company_size: 'varchar',
    industry: 'varchar',
    budget_range: 'varchar',
    decision_timeline: 'varchar',
    
    // Engagement Tracking
    email_opens: 'number',
    email_clicks: 'number',
    website_visits: 'number',
    last_engagement: 'date'
};

// Lead Scoring Algorithm
function calculateLeadScore(person) {
    let score = 0;
    
    // Assessment Score (40% weight)
    const assessmentScore = person.custom_fields.assessment_score || 0;
    if (assessmentScore >= 80) score += 40;
    else if (assessmentScore >= 60) score += 30;
    else if (assessmentScore >= 40) score += 20;
    else score += 10;
    
    // Company Size (25% weight)
    const companySize = person.custom_fields.company_size;
    if (companySize === '500+ medewerkers') score += 25;
    else if (companySize === '100-500 medewerkers') score += 20;
    else if (companySize === '50-100 medewerkers') score += 15;
    else score += 10;
    
    // Engagement Level (20% weight)
    const emailOpens = person.custom_fields.email_opens || 0;
    const emailClicks = person.custom_fields.email_clicks || 0;
    const engagementScore = (emailOpens * 2) + (emailClicks * 5);
    if (engagementScore >= 20) score += 20;
    else if (engagementScore >= 10) score += 15;
    else if (engagementScore >= 5) score += 10;
    else score += 5;
    
    // Decision Timeline (15% weight)
    const timeline = person.custom_fields.decision_timeline;
    if (timeline === 'Binnen 1 maand') score += 15;
    else if (timeline === '1-3 maanden') score += 12;
    else if (timeline === '3-6 maanden') score += 8;
    else score += 5;
    
    return Math.min(score, 100);
}

// Priority Classification
function getPriorityLevel(leadScore) {
    if (leadScore >= 80) return 'Hot';
    if (leadScore >= 60) return 'Warm';
    return 'Cold';
}
```
